Getting started

How to install it? 

composer require xudid/ui
How to use it ? create a php file with the composer autoloader 
require "../vendor/autoload.php";
use Ui\Views\Page;
use Ui\Widgets\Table\{
                        DivRowTable,
                        TableLegend,
                        TableColumn
                     };

A little example


require "../vendor/autoload.php";

use Ui\Views\Page;

use Ui\Widgets\Table\{
                        DivRowTable,
                        TableLegend,
                        TableColumn
                     };

$css = "/css/style.css";

$legend = new TableLegend(
                            "Participants",
                             TableLegend::TOP_LEFT
                         );


$table = new DivRowTable(
             [$legend],
             [
                new TableColumn("name","Nom"),
                new TableColumn("pseudo","Alias"),
                new TableColumn("age","Age")
             ],
             [
                [
                    "name" => "Arthurus",
                    "pseudo" => "Arthure",
                    "age" => "33"
                ],
                ["name" => "Genievre",
                 "pseudo" => "Geny",
                 "age" => "31"
                ],
                ["name" => "Yvain",
                 "pseudo" => "Lion,
                 "age" => "18 ans "
                ]
             ],
             false,
             "/"
            );

$page = (new Page())->setTitle("Ma Page de Test");
$page->body()->addElement($table);
    
echo $page;
